#If the hospital increases its admission capacity by 10%, how will this affect the measures of central tendency?

import numpy as np
from scipy import stats

# Patient admission data for the past 10 days
admissions = [32, 28, 35, 30, 29, 27, 31, 34, 33, 30]

# Calculate the mean, median, and mode
mean_admission = np.mean(admissions)
median_admission = np.median(admissions)
mode_result = stats.mode(admissions)

# Directly access mode and count
mode_admission = mode_result.mode  # The mode value (scalar)
mode_count = mode_result.count  # The count of the mode (scalar)

# Display the original measures of central tendency
print(f"Original Data: {admissions}")
print(f"Mean: {mean_admission}")
print(f"Median: {median_admission}")
print(f"Mode: {mode_admission} (appears {mode_count} times)" if mode_admission else "Mode: No mode found")

# Increase the admission capacity by 10%
increased_admissions = [admission * 1.10 for admission in admissions]

# Calculate the new mean, median, and mode after the increase
mean_increased = np.mean(increased_admissions)
median_increased = np.median(increased_admissions)
mode_result_increased = stats.mode(increased_admissions)

# Directly access new mode and count
mode_increased = mode_result_increased.mode  # The new mode value (scalar)
mode_count_increased = mode_result_increased.count  # The new count of the mode (scalar)

# Display the new measures of central tendency
print(f"\nData After 10% Increase: {increased_admissions}")
print(f"New Mean: {mean_increased}")
print(f"New Median: {median_increased}")
print(f"New Mode: {mode_increased} (appears {mode_count_increased} times)" if mode_increased else "New Mode: No mode found")

# Interpretation of results
print("\nInterpretation:")
print("1. The mean is affected by every value, so it will increase with the 10% rise in patient admissions.")
print("2. The median will change if the central value shifts due to the increased capacity.")
print("3. The mode may not change if the relative frequency of admissions remains the same after the increase.")

#Section B -5. The hospital claims that the average patient wait time in the emergency department is 30 minutes. A sample of 10 patient wait times (in minutes) is recorded:
#[32, 29, 31, 34, 33, 27, 30, 28, 35, 26]
#Test whether the hospital’s claim is valid at a 5% significance level.
#State the null and alternative hypotheses.
#If the wait time significantly exceeds 30 minutes, what changes should the hospital implement to reduce waiting time?

import numpy as np
from scipy import stats

# Sample of 10 patient wait times in minutes
wait_times = [32, 29, 31, 34, 33, 27, 30, 28, 35, 26]

# Sample size
n = len(wait_times)

# Calculate sample mean and sample standard deviation
sample_mean = np.mean(wait_times)
sample_std = np.std(wait_times, ddof=1)  # Using ddof=1 for sample standard deviation

# Population mean (claimed average)
population_mean = 30

# Compute t-statistic
t_statistic = (sample_mean - population_mean) / (sample_std / np.sqrt(n))

# Degrees of freedom
df = n - 1

# Find the critical t-value for a two-tailed test at 5% significance level
alpha = 0.05
critical_t = stats.t.ppf(1 - alpha/2, df)

# Compare the t-statistic with the critical t-value
print(f"Sample Mean: {sample_mean}")
print(f"Sample Standard Deviation: {sample_std}")
print(f"T-statistic: {t_statistic}")
print(f"Critical t-value: {critical_t}")

# Decision: Reject or Fail to Reject Null Hypothesis
if abs(t_statistic) > critical_t:
    print("\nReject the null hypothesis. The average wait time is significantly different from 30 minutes.")
else:
    print("\nFail to reject the null hypothesis. There is no significant difference from 30 minutes.")
